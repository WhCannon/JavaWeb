```html
<html>
 <head>
  <title>HTML示例</title>
 </head>
 <body>
	<!-- 列表标签 -->
	<dl>

		<li>学工部</li>
		<li>人事部</li>
	</ol>

	<hr/>
	<!-- 无序列表 -->

		<li>财务部</li>
		<li>学工部</li>
		<li>人事部</li>
	</ul>
 </body>
</html>
```
